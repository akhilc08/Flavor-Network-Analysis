BCE link prediction pattern (used for both molecular and recipe losses):
```python
from torch_geometric.utils import negative_sampling

def link_pred_loss(z_src, z_dst, pos_edge_index, num_nodes, device):
    """
    z_src, z_dst: (N, embed_dim) — embeddings of source and dest node types
    pos_edge_index: (2, K) — indices of positive edges in this split
    num_nodes: int or (int, int) for bipartite
    """
    src_pos = z_src[pos_edge_index[0]]
    dst_pos = z_dst[pos_edge_index[1]]
    pos_scores = (src_pos * dst_pos).sum(dim=-1)

    neg_edge_index = negative_sampling(
        edge_index=pos_edge_index,
        num_nodes=num_nodes,
        num_neg_samples=pos_edge_index.size(1),
    )
    src_neg = z_src[neg_edge_index[0]]
    dst_neg = z_dst[neg_edge_index[1]]
    neg_scores = (src_neg * dst_neg).sum(dim=-1)

    scores = torch.cat([pos_scores, neg_scores])
    labels = torch.cat([
        torch.ones(pos_scores.size(0)),
        torch.zeros(neg_scores.size(0)),
    ]).to(device)
    return F.binary_cross_entropy_with_logits(scores, labels)
```

InfoNCE pattern (L2-normalize INSIDE function, not in model forward):
```python
def info_nce_loss(z: torch.Tensor, pos_pairs: torch.Tensor, tau: float) -> torch.Tensor:
    """
    z: (N, embed_dim) ingredient embeddings — NOT pre-normalized
    pos_pairs: (2, K) index pairs; pos_pairs[0] = anchors, pos_pairs[1] = positives
    tau: temperature (default 0.15)
    """
    z_norm = F.normalize(z, dim=-1)
    sim = torch.mm(z_norm, z_norm.T) / tau
    mask = torch.eye(z_norm.size(0), dtype=torch.bool, device=z_norm.device)
    sim.masked_fill_(mask, float('-inf'))
    labels = pos_pairs[1]
    loss = F.cross_entropy(sim[pos_pairs[0]], labels)
    return loss
```

Combined loss:
```python
def combined_loss(mol_loss, rec_loss, nce_loss, alpha, beta, gamma):
    return alpha * mol_loss + beta * rec_loss + gamma * nce_loss
```
