Embedding export pattern (CPU-first to avoid MPS double-spike OOM):
```python
@torch.no_grad()
def export_embeddings(model, data, ingredient_id_to_idx, embed_path, device):
    model.eval()
    data_cpu = data.to('cpu')
    model_cpu = model.to('cpu')
    z_dict = model_cpu(data_cpu.x_dict, data_cpu.edge_index_dict)
    z_ing = z_dict['ingredient'].numpy()
    idx_to_id = {v: k for k, v in ingredient_id_to_idx.items()}
    embeddings = {idx_to_id[i]: z_ing[i] for i in range(len(z_ing))}
    Path(embed_path).parent.mkdir(parents=True, exist_ok=True)
    with open(embed_path, 'wb') as f:
        pickle.dump(embeddings, f)
    model.to(device)
```

def test_checkpoint_save_on_improvement(tmp_path):
    from model.train_gat import save_checkpoint
    ckpt_path = tmp_path / "best_model.pt"
    state = {'model_state_dict': {}, 'epoch': 1, 'best_auc': 0.75}

    # Save when AUC improves
    best_auc = 0.70
    new_auc = 0.75
    if new_auc > best_auc:
        save_checkpoint(state, ckpt_path)
    assert ckpt_path.exists(), "Checkpoint should be saved when AUC improves"

    # Should NOT save when AUC declines
    ckpt_path2 = tmp_path / "should_not_exist.pt"
    best_auc = 0.75
    new_auc = 0.70
    if new_auc > best_auc:
        save_checkpoint(state, ckpt_path2)
